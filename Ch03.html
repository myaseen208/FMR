3  Building a Portfolio

3.1 Portfolio Returns

Example

Now, we collect individual returns into a portfolio, which is a weighted set of asset returns. Accordingly, the first thing we need to do is assign a weight to each asset. Recall that our portfolio will have the following asset mix:

  1. SPY (S&P500 ETF) weighted 25%

  2. EFA (a non-US equities ETF) weighted 25%

  3. IJS (a small-cap value ETF) weighted 20%

  4. EEM (an emerging-markets ETF) weighted 20%

  5. AGG (a bond ETF) weighted 10%

symbols <- c("SPY","EFA", "IJS", "EEM","AGG")

prices <- 
  getSymbols(
      Symbols     = symbols
    , src         = 'yahoo'
    , from        = "2012-12-31"
    , to          = "2017-12-31"
    , auto.assign = TRUE
    , warnings    = FALSE
    ) %>% 
  map(
    .x =,
    .f = ~Ad(get(.))
    ) %>% 
  reduce(merge) %>%
  `colnames<-`(symbols)

prices_monthly <- 
  to.monthly(
      x         = prices
    , indexAt   = "lastof"
    , drop.time = TRUE
    , OHLC      = FALSE
    )

asset_returns_xts <- 
  Return.calculate(
        prices = prices_monthly
      , method = c("discrete", "log", "difference")[2]
      ) %>% 
  na.omit()

w <- c(0.25, 0.25, 0.20, 0.20, 0.10)

portfolio_returns_xts_rebalanced_monthly <- 
  Return.portfolio(asset_returns_xts, weights = w, rebalance_on = "months") %>%
  `colnames<-`("returns")

Plot2.1 <- 
  highchart(type = "stock") %>%
  hc_title(text = "Portfolio Monthly Returns") %>%
  hc_add_series(portfolio_returns_xts_rebalanced_monthly$returns,
                name = "Rebalanced Monthly",
                color = "cornflowerblue") %>%
  hc_add_theme(hc_theme_flat()) %>%
  hc_navigator(enabled = FALSE) %>%
  hc_scrollbar(enabled = FALSE) %>%
  hc_legend(enabled = TRUE) %>%
  hc_exporting(enabled = TRUE)
